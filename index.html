安装

# install
yarn global add vuepress@next
# OR npm install -g vuepress@next

# create a markdown file
echo '# Hello VuePress' > README.md

# start writing
vuepress dev

# build to static files
vuepress build

# npx
npx vuepress dev docs

# 使用 yarn
yarn dev

yarn build
